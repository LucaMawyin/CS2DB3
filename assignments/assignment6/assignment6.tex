\subsubsection{}

\begin{enumerate}
    \item The attribute \textit{rid}, which represents the recipe idetifier, determines all other attributes of the recipe. If two tuples agree on the value for attribute \textit{rid}, then they agree on values for attributes \textit{rid, title, total-duration, min-price}.
    \item The value for attribute \textit{min-price}, which represents the minimum price of a recpie, can be computed using the unit price of the cheapest prduct offering of each ingredient and the amount of that ingredient required by the recipe.
    \item The tuple of attributes \textit{rid, step-order}, which represents the order of a step within a recipe, determines all other step attributes \textit{rid, step-order, description, duration}.
    \item \textit{step-order} does not uniquely determine \textit{ingredient}, as a step can involve multiple ingredients. Therefore, a recipe may have many steps $n$ with the same \textit{step-order} and different \textit{ingredient} values. As different attributes of an \textit{ingredient} can be used in a step, the attribute \textit{amount} is not uniquely determined by \textit{step-order}. Therefore, the combination of attributes \textit{step-order} and \textit{ingredient} determines the attribute \textit{amount} within the context of a specific recipe with attribute \textit{rid}.
    \item The attribute \textit{UPC}, which represents the unique product identifier code of an \textit{ingredient} offering, determines all other attributes of offering \textit{UPC, ingredient, producer, quantity, price, unit-price}. Attribute \textit{ingredient} can have multple values of \textit{UPC}, as many producers can produce the same ingredient. Furthermore, producer can have multiple values of attribute \textit{UPC}, as a producer can produce many different ingredient offerings.
    \item The value for attribute \textit{unit-price}, which represents the price per unit of an ingredient, can be computed by dividing the price of an ingredient offering by the quantity of such offering.
\end{enumerate}

\begin{center}
    If $x$ is an attribute, then $x\rightarrow x$
\end{center}
\begin{enumerate}
    \item \textit{Reflexivity:} If $A \subseteq B$, then $B \rightarrow A$
    \item Let $A = x$ and $B = x$
    \item $x = x \subseteq x$ holds trivially
    \item Because $x \subseteq x$, $x \rightarrow x$ holds by Armstrong Axiom of Reflexivity.
\end{enumerate}

\subsubsection{}
\begin{center}
    If $A \rightarrow B$ and $C \rightarrow D$, then $AC \rightarrow BD$
\end{center}
\begin{enumerate}
    \item Let $\tau_1,\tau_2\in\mathcal{J}(\mathbf{R})$
    \item Assume the antecendent $\tau_1[AC]=\tau_2[AC]\equiv\tau_1[A]=\tau_2[A]\land\tau_1[C]=\tau_2[C]$
    \item Prove the consequent $\tau_1[BD]=\tau_2[BD]$
    \item By the antecedent and functional dependency $A\rightarrow B$, we conclude $\tau_1[B]=\tau_2[B]$
    \item By the antecedent and functional dependency $C\rightarrow D$, we conclude $\tau_1[D]=\tau_2[D]$
    \item Therefore we have $\big(\tau_1[B]=\tau_2[B]\big)\land\big(\tau_1[D]=\tau_2[D]\big)\equiv\tau_1[BD]=\tau_2[BD]$
\end{enumerate}

\subsubsection{}
\begin{center}
    If $A\rightarrow B$ and $C \rightarrow D$, then $\big(A\cup\big(C~\backslash~B\big)\big)\rightarrow D$
\end{center}
\begin{enumerate}
    \item First define $C$ as $C=(C\cap B) \cup (C~\backslash~B)$
    \item As $(C~\backslash~B)\subseteq C$, using Armstrong axiom of \textit{Reflexivity}, then $C \rightarrow (C~\backslash~B)$
    \item Using Union, as $C \rightarrow (C~\backslash~B)$ and $C\rightarrow D$, then $C \rightarrow D\cup(C~\backslash~B)$
    \item As $(C\cap B)\subseteq B$, we can say that $(C\cap B)\cup(C~\backslash~B)\subseteq B\cup(C~\backslash~B)$
    \item Taking our definition of $C$, we can replace it to get $C\subseteq B\cup(C~\backslash~B)$
    \item Using Armstrong axiom of \textit{Reflexivity}, as $C\subseteq B\cup(C~\backslash~B)$, then $B\cup(C~\backslash~B)\rightarrow C$ 
    \item Now we take $A\rightarrow B$
    \item Using Armstrong axiom \textit{Augmentation}, because $A\rightarrow B$, then $A\cup(C~\backslash~B)\rightarrow B\cup(C~\backslash~B)$
    \item Using Armstrong axiom \textit{Transitivity}, because $B\cup(C~\backslash~B)\rightarrow C$ and $C \rightarrow D\cup(C~\backslash~B)$, then \newline $B\cup(C~\backslash~B)\rightarrow D\cup(C~\backslash~B)$
    \item Using Armstrong axiom \textit{Transitivity}, because $A\cup(C~\backslash~B)\rightarrow B\cup(C~\backslash~B)$ and $B\cup(C~\backslash~B)\rightarrow D\cup(C~\backslash~B)$, then \newline $A\cup(C~\backslash~B)\rightarrow D\cup(C~\backslash~B)$
    \item Finally, using Decomposition, because $A\cup(C~\backslash~B)\rightarrow D\cup(C~\backslash~B)$, then $A\cup(C~\backslash~B)\rightarrow D$
\end{enumerate}

    If $A\rightarrow B$ and $D\subseteq(B\cup C)$, then $AC\rightarrow D$
\end{center}
\begin{enumerate}
    \item Using Armstrong axiom \textit{Reflexivity}, because $D\subseteq(B\cup C)$, then $(B\cup C)\rightarrow D$, or $BC \rightarrow D$
    \item Taking $A\rightarrow B$ and using Armstrong axiom \textit{Augmentation}, we get $AC\rightarrow BC$
    \item Using Armstrong axiom \textit{Transitivity}, because $AC\rightarrow BC$ and $BC \rightarrow D$, then $AC\rightarrow D$
\end{enumerate}
